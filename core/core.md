# 🔧 Модуль 1: init_system()

```python
def init_system(self):
    """
    Инициализирует все ключевые подсистемы ядра SAI.
    Запускается один раз при создании экземпляра или перезапуске системы.
    """
    from commands import CommandRegistry
    from roles import RoleRegistry
    from thinking import Flow, A1, Outflow
    from corcop import CoRRunner, CoPExecutor
    from response_template import StructuredResponse
    from style import StyleManager
    from feedback import FeedbackEngine
    from search import SearchAgent
    from intent import IntentVectorizer, MirrorEngine, GoalBuilder

    self.commands = CommandRegistry()
    self.roles = RoleRegistry()
    self.flow = Flow()
    self.a1 = A1()
    self.outflow = Outflow()
    self.cor = CoRRunner()
    self.cop = CoPExecutor()
    self.feedback = FeedbackEngine()
    self.style = StyleManager()
    self.intent = IntentVectorizer()
    self.mirror = MirrorEngine()
    self.goal_builder = GoalBuilder()
    self.search = SearchAgent(enabled=False)
```

# 🔧 Модуль 2: handle_request(prompt)

```python
def handle_request(self, prompt: str):
    """
    Главный обработчик входящего запроса.
    Пропускает ввод через весь цикл мышления:
    анализ → CoR → CoP → роли → стиль → фидбек → возврат результата.
    """
    self.analyze_input(prompt)
    self.generate_CoR()
    self.run_CoP()
    self.apply_roles()
    self.style_response()
    self.log_feedback()
    return self.return_structured()
```

# 🔧 Модуль 3: analyze_input(prompt)

```python
def analyze_input(self, prompt: str):
    """
    Анализирует входной запрос:
    - выявляет намерение, мотивацию, подповерхностный смысл
    - строит когнитивное зеркало пользователя
    - извлекает цель и формулирует начальное направление
    - запускает структурный и логический анализ текста
    """
    self.trace = {
        "intent": self.intent.detect(prompt),
        "mirror": self.mirror.reflect(prompt),
        "goal": self.goal_builder.construct(prompt)
    }
    self.context = self.flow.analyze(prompt) + self.a1.check(prompt)
```

# 🔧 Модуль 4: generate_CoR()

```python
def generate_CoR(self):
    """
    Строит цепь рассуждений (Chain of Reason) на основе анализа запроса.
    Эта цепь определяет:
    - направление логики
    - стратегию ответа
    - скрытые конфликты и желания
    - эмоциональные и смысловые пласты
    """
    self.cor_map = self.cor.build(self.context, self.trace)
```

# 🔧 Модуль 5: run_CoP()

```python
def run_CoP(self):
    """
    Запускает CoP (Chain of Prompt Execution):
    - проходит по шагам CoR
    - вызывает роли и мышление на каждом этапе
    - собирает последовательность рассуждений
    """
    selected_roles = self.roles.select_for(self.context)
    self.steps = self.cop.execute(self.cor_map, selected_roles)
```

# 🔧 Модуль 6: apply_roles()

```python
def apply_roles(self):
    """
    Применяет выбранные роли к текущему контексту:
    - активирует поведенческие маски (JLO, AGOD, Strategist и др.)
    - модифицирует стиль, логику или глубину мышления
    - позволяет ролям интерпретировать или трансформировать CoR/шаги
    """
    self.active_roles = self.roles.select_for(self.context)
    for role in self.active_roles:
        role.use(self.context, self.trace, self.cor_map)
```

# 🔧 Модуль 7: style_response()

```python

def style_response(self):
    """
    Применяет выбранный стиль оформления к финальному выводу:
    - адаптирует под тональность, эмоциональный контекст и роль
    - форматирует последний шаг мышления в читаемый результат
    """
    final_step = self.steps[-1] if self.steps else "Нет результата."
    self.active_style = self.style.pick(self.trace, roles=self.active_roles)
    self.result = self.active_style.apply(final_step)
```

# 🔧 Модуль 8: log_feedback()

```python
def log_feedback(self):
    """
    Сохраняет текущий опыт взаимодействия:
    - записывает результат, CoR, эмоции, стиль и цель
    - отправляет их в feedback-систему для обучения и последующей адаптации
    """
    self.feedback.record(
        result=self.result,
        trace=self.trace,
        context=self.context,
        cor=self.cor_map,
        steps=self.steps
    )
```

# 🔧 Модуль 9: return_structured()

```python

def return_structured(self):
    """
    Возвращает финальный ответ в виде структурированного объекта:
    - включает CoR, шаги рассуждения, финальный результат, стиль и трассировку
    - обеспечивает стандартизированный формат для отображения и анализа
    """
    return StructuredResponse(
        cor=self.cor_map,
        logic_steps=self.steps,
        output=self.result,
        style=self.active_style,
        trace=self.trace
    )
```

# 🔧 Модуль 10: enable_search() / disable_search()

```python
def enable_search(self):
    """
    Включает доступ к внешнему поисковому агенту.
    Используется в задачах, где требуется внешнее знание.
    """
    self.search.enabled = True

def disable_search(self):
    """
    Отключает внешний поисковый агент.
    Используется для полной автономии мышления SAI.
    """
    self.search.enabled = False
```
