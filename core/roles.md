```python
        self.empathy = empathy
        self.irony_density = irony_density

        trace["JAI_mode"] = {
            "snark": self.snark,
```

```python
        self.depth = depth
        self.scenario_mode = scenario_mode
        self.clarity_boost = clarity_boost

    def can_handle(self, context: dict) -> bool:
        """
        Включается при запросах вида:
        - "что делать, если..."

    def use(self, context: dict, trace: dict, cor: dict):
        trace["Strategist_mode"] = {
            "depth": self.depth,
            "scenario_mode": self.scenario_mode,
            "clarity_boost": self.clarity_boost
        }

    def respond(self, content: str) -> str:
        header = "🧭 Стратегический обзор:\n"

            options = "- Вариант A: осторожный путь\n- Вариант B: агрессивный рост\n- Вариант C: уход в неизвестность"
            return f"{header}{clarity}\n{options}"
```

```python
class HybridRole(BaseRole):
    """
    HybridRole — композиционная над-роль.
    Она объединяет поведение нескольких совместимых ролей,
    отфильтровывая конфликтующие и взвешивая вклад каждой в финальный результат.

    Механизм:
    - собирает все роли, которые прошли `can_handle(context)`
    - оценивает их релевантность (вклад)
    - устраняет конфликты (например, Master + Wrath могут конфликтовать)
    - формирует синтетический ответ на основе веса

    Под капотом работает алгоритм согласования ролей через контекстный scoring.
    """

    name = "HybridRole"
    description = "Мета-объединитель. Формирует составной профиль поведения из релевантных ролей."

    def __init__(self, roles: list[BaseRole]):
        self.roles = self._filter_conflicts(roles)

    def _role_score(self, role: BaseRole, context: dict) -> float:
        """
        Простейший скоринг на основе:
        - наличия ключевых слов
        - уровня конфигурации (если кастомизирован — выше)
        - контекстной плотности (raw, trace)
        """
        base = 0.5
        if role.can_handle(context):
            base += 0.3
        if hasattr(role, '__dict__') and len(role.__dict__) > 3:
            base += 0.2
        return base

    def _filter_conflicts(self, roles: list[BaseRole]) -> list[BaseRole]:
        """
        Удаляет заведомо конфликтующие комбинации:
        - Wrath + Dreamer
        - AGOD + JLO (если уровень elevation > 0.9)
        - Master + любая роль с dominance = high
        """
        filtered = []
        names = [r.name for r in roles]

        for role in roles:
            if role.name == "Wrath" and "Dreamer" in names:
                continue
            if role.name == "AGOD" and "JLO" in names and getattr(role, 'elevation', 0) > 0.9:
                continue
            if role.name == "Master" and any(getattr(r, 'dominance', 0) > 0.8 for r in roles if r != role):
                continue
            filtered.append(role)
        return filtered

    def use(self, context: dict, trace: dict, cor: dict):
        self.weights = {}
        for role in self.roles:
            score = self._role_score(role, context)
            self.weights[role.name] = score
            role.use(context, trace, cor)
        trace["HybridRole_weights"] = self.weights

    def respond(self, content: str) -> str:
        # Собираем вклад всех ролей
        responses = []
        for role in self.roles:
            part = role.respond(content)
            weight = self.weights.get(role.name, 0.5)
            responses.append((weight, part))

        # Сортируем по значимости, собираем
        sorted_parts = sorted(responses, key=lambda x: -x[0])
        combined = "\n---\n".join(p for _, p in sorted_parts)

        return f"[HYBRID ROLE RESPONSE]\n{combined}"
```

```python
class RoleRegistry:
    """
    RoleRegistry — центральный регистр всех доступных ролей в SAI.
    Хранит список активных ролей, позволяет их регистрировать, получать и фильтровать.

    Возможности:
    - автоматическая инициализация встроенных ролей
    - доступ по имени
    - проверка доступности в контексте
    - возвращает список ролей, подходящих под задачу
    """

    def __init__(self):
        self._roles = {}
        self._initialize_default_roles()

    def _initialize_default_roles(self):
        """
        Регистрирует все встроенные роли.
        """
        self.register(JLO())
        self.register(JAI())
        self.register(Strategist())
        self.register(AGOD())
        self.register(Dreamer())
        self.register(Wrath())
        self.register(Master())
        self.register(BaseRole())  # fallback

    def register(self, role: BaseRole):
        """
        Добавляет роль в реестр.
        """
        self._roles[role.name] = role

    def get(self, name: str) -> BaseRole:
        """
        Возвращает роль по имени, если она зарегистрирована.
        """
        return self._roles.get(name)

    def all(self) -> list[BaseRole]:
        """
        Возвращает список всех ролей.
        """
        return list(self._roles.values())

    def applicable(self, context: dict) -> list[BaseRole]:
        """
        Возвращает только те роли, которые могут обработать текущий запрос.
        """
        return [role for role in self._roles.values() if role.can_handle(context)]
```

```python
def select_for(context: dict) -> BaseRole:
    """
    Автоматически выбирает релевантную роль или формирует гибридную (HybridRole),
    основываясь на текущем контексте и состоянии запроса.

    Логика:
    - Проверяются все зарегистрированные роли через `can_handle()`
    - Если несколько подходят — строится `HybridRole` из них
    - Если нет подходящих — возвращается `BaseRole` (нулевая позиция)
    """

    registry = RoleRegistry()
    matching_roles = registry.applicable(context)

    if not matching_roles:
        return BaseRole()

    if len(matching_roles) == 1:
        return matching_roles[0]

    return HybridRole(matching_roles)
```
